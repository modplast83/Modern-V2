    <script>
        let scene, camera, renderer, controls;
        let bagGroup, bagMesh, printMesh, handles = [];
        let printTexture;

        const SCALE = 10;

        init();
        animate();

        function init() {
            const container = document.getElementById('canvas-container');

            scene = new THREE.Scene();
            scene.background = new THREE.Color('#f1f5f9');

            camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 100);
            camera.position.set(6, 3, 8);

            renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(window.devicePixelRatio);
            renderer.shadowMap.enabled = true;
            renderer.shadowMap.type = THREE.PCFSoftShadowMap;
            container.appendChild(renderer.domElement);

            controls = new THREE.OrbitControls(camera, renderer.domElement);
            controls.enableDamping = true;
            controls.dampingFactor = 0.05;
            controls.target.set(0, 2, 0);
            controls.minDistance = 2;
            controls.maxDistance = 15;

            setupLighting();
            setupFloor();
            
            bagGroup = new THREE.Group();
            scene.add(bagGroup);

            // إنشاء نسيج الطباعة الأولي
            updatePrintCanvas();
            
            updateBagGeometry();
            setupUIListeners();
            window.addEventListener('resize', onWindowResize);
        }

        function setupLighting() {
            const ambient = new THREE.AmbientLight(0xffffff, 0.6);
            scene.add(ambient);

            const dirLight = new THREE.DirectionalLight(0xffffff, 1.0);
            dirLight.position.set(5, 10, 7);
            dirLight.castShadow = true;
            dirLight.shadow.mapSize.width = 2048;
            dirLight.shadow.mapSize.height = 2048;
            scene.add(dirLight);

            const pointLight = new THREE.PointLight(0xe0f2fe, 0.5);
            pointLight.position.set(-5, 5, 5);
            scene.add(pointLight);
        }

        function setupFloor() {
            const floorGeo = new THREE.PlaneGeometry(50, 50);
            const floorMat = new THREE.MeshStandardMaterial({ color: '#e2e8f0', roughness: 0.5 });
            const floor = new THREE.Mesh(floorGeo, floorMat);
            floor.rotation.x = -Math.PI / 2;
            floor.receiveShadow = true;
            scene.add(floor);
            
            const grid = new THREE.GridHelper(50, 50, 0xcbd5e1, 0xf1f5f9);
            grid.position.y = 0.01;
            scene.add(grid);
        }

        // إنشاء خامة البلاستيك
        function getPlasticMaterial(type, colorHex) {
            let opts = {
                color: new THREE.Color(colorHex),
                side: THREE.DoubleSide,
                transparent: true,
            };

            if (type === 'ldpe') {
                opts.roughness = 0.1;
                opts.metalness = 0.1;
                opts.transmission = 0.6; // شفافية
                opts.opacity = 0.9;
                opts.clearcoat = 1.0;
            } else {
                opts.roughness = 0.6;
                opts.metalness = 0.0;
                opts.transmission = 0.0; // معتم
                opts.opacity = 1.0;
            }
            return new THREE.MeshPhysicalMaterial(opts);
        }

        // رسم محتوى الطباعة (نظام فلكسو)
        function updatePrintCanvas() {
            const canvas = document.getElementById('print-canvas');
            const ctx = canvas.getContext('2d');
            const text = document.getElementById('print-text').value;
            
            // تنظيف الكانفاس (شفاف)
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            // قراءة الألوان المفعلة
            const colors = [];
            for(let i=1; i<=4; i++) {
                if(document.getElementById('enable-c' + i).checked) {
                    colors.push(document.getElementById('color-' + i).value);
                }
            }

            if(text && colors.length > 0) {
                ctx.textAlign = 'center';
                ctx.textBaseline = 'middle';
                const cx = canvas.width / 2;
                const cy = canvas.height / 2;

                // محاكاة فرز الألوان (تراكب الألوان في الفلكسو)
                // اللون الأول سيكون الظل / الخلفية
                if(colors.length > 1) {
                    ctx.font = 'bold 130px Tajawal, sans-serif';
                    ctx.fillStyle = colors[1];
                    ctx.fillText(text, cx + 8, cy + 8);
                }
                
                // اللون الثاني إطار
                if(colors.length > 2) {
                    ctx.font = 'bold 120px Tajawal, sans-serif';
                    ctx.strokeStyle = colors[2];
                    ctx.lineWidth = 15;
                    ctx.strokeText(text, cx, cy);
                }

                // اللون الأساسي (الأول دائماً للملء)
                ctx.font = 'bold 120px Tajawal, sans-serif';
                ctx.fillStyle = colors[0];
                ctx.fillText(text, cx, cy);

                // أشكال إضافية لإظهار الألوان الباقية
                if(colors.length > 3) {
                    ctx.fillStyle = colors[3];
                    ctx.beginPath();
                    ctx.arc(cx, cy - 150, 40, 0, Math.PI * 2);
                    ctx.fill();
                }
            }

            if(!printTexture) {
                printTexture = new THREE.CanvasTexture(canvas);
                printTexture.anisotropy = renderer.capabilities.getMaxAnisotropy();
            } else {
                printTexture.needsUpdate = true;
            }
        }

        function updateBagGeometry() {
            // تنظيف الجروب
            while(bagGroup.children.length > 0){ 
                const child = bagGroup.children[0];
                bagGroup.remove(child); 
            }
            handles = [];

            const w = parseFloat(document.getElementById('width').value) / SCALE;
            const h = parseFloat(document.getElementById('height').value) / SCALE;
            let d = parseFloat(document.getElementById('depth').value) / SCALE;
            if (d === 0) d = 0.05;

            const handleType = document.getElementById('handle').value;
            const plasticType = document.getElementById('type').value;
            const colorHex = document.getElementById('bag-color').value;

            const material = getPlasticMaterial(plasticType, colorHex);

            // 1. هندسة الكيس الأساسية
            let geo = new THREE.BoxGeometry(w, h, d, 32, 32, 8);
            geo.translate(0, h/2, 0);

            // تشويه الهندسة لتبدو ككيس
            const pos = geo.getAttribute('position');
            for(let i = 0; i < pos.count; i++) {
                let x = pos.getX(i);
                let y = pos.getY(i);
                let z = pos.getZ(i);
                const yNorm = y / h;

                // طية الأجناب (Gusset)
                if (Math.abs(x) > (w/2 - 0.02) && Math.abs(z) < (d/4)) {
                    x -= Math.sign(x) * (d * 0.2);
                }

                // تشكيل بناءً على نوع المقبض
                if (handleType === 'vest') {
                    // كيس علاقي: المنتصف العلوي فارغ (ينزل للأسفل)
                    if (yNorm > 0.7 && Math.abs(x) < w * 0.25) {
                        const dip = (yNorm - 0.7) * 2;
                        y -= dip;
                    }
                } else {
                    // الأكياس العادية تنضغط من الأعلى
                    if (yNorm > 0.8) {
                       const pinch = (yNorm - 0.8) * 5;
                       z *= (1 - pinch * 0.8);
                    }
                }

                pos.setXYZ(i, x, y, z);
            }
            geo.computeVertexNormals();

            bagMesh = new THREE.Mesh(geo, material);
            bagMesh.castShadow = true;
            bagMesh.receiveShadow = true;
            bagGroup.add(bagMesh);

            // 2. إضافة المقابض الإضافية (بنانة أو Loop)
            if (handleType === 'banana') {
                // محاكاة فتحة البنانة برسم توروس (Torus) داكن صغير في الأعلى
                const holeGeo = new THREE.TorusGeometry(w * 0.1, 0.03, 16, 32);
                holeGeo.scale(1, 0.4, 1); // جعله بيضاوياً (كدني)
                const holeMat = new THREE.MeshBasicMaterial({ color: 0x222222 }); // يمثل الفراغ
                
                const holeFront = new THREE.Mesh(holeGeo, holeMat);
                holeFront.position.set(0, h * 0.85, d/2 * 0.2); // منضغط من الأعلى
                
                const holeBack = holeFront.clone();
                holeBack.position.set(0, h * 0.85, -d/2 * 0.2);

                bagGroup.add(holeFront, holeBack);
            } 
            else if (handleType === 'loop') {
                // مقبض ملحوم خارجي
                const handleWidth = w * 0.25;
                const handleGeo = new THREE.TorusGeometry(handleWidth, 0.06, 16, 32, Math.PI);
                
                const h1 = new THREE.Mesh(handleGeo, material);
                h1.position.set(0, h * 0.98, d/2 * 0.2);
                
                const h2 = new THREE.Mesh(handleGeo, material);
                h2.position.set(0, h * 0.98, -d/2 * 0.2);

                bagGroup.add(h1, h2);
            }

            // 3. طبقة الطباعة (Decal / Print Mesh)
            // لكي تظهر الطباعة بوضوح فوق البلاستيك الشفاف، ننشئ سطحاً خفياً يحمل نسيج الطباعة
            const printMat = new THREE.MeshBasicMaterial({
                map: printTexture,
                transparent: true,
                depthWrite: false, // لمنع تعارض العمق مع البلاستيك الشفاف
                polygonOffset: true,
                polygonOffsetFactor: -1,
                polygonOffsetUnits: -1
            });
            
            // تكرار هندسة الكيس للطباعة ولكن وتكبيرها قلييييلاً جداً لتكون فوق البلاستيك
            const printGeo = geo.clone();
            printMesh = new THREE.Mesh(printGeo, printMat);
            // تكبير طفيف جداً
            printMesh.scale.set(1.002, 1.002, 1.002);
            bagGroup.add(printMesh);

            controls.target.set(0, h/2, 0);
        }

        function setupUIListeners() {
            // تحديث الأبعاد
            ['width', 'height', 'depth'].forEach(id => {
                document.getElementById(id).addEventListener('input', (e) => {
                    document.getElementById(`${id}-val`).innerText = e.target.value;
                    updateBagGeometry();
                });
            });

            // تحديث النوع والشكل
            ['handle', 'type', 'bag-color'].forEach(id => {
                document.getElementById(id).addEventListener('change', updateBagGeometry);
            });

            // تحديث الطباعة (زر منفصل للأداء)
            document.getElementById('update-print-btn').addEventListener('click', () => {
                updatePrintCanvas();
            });

            // UI toggles
            const panel = document.getElementById('control-panel');
            const openBtn = document.getElementById('toggle-panel-btn');
            document.getElementById('close-panel-btn').addEventListener('click', () => {
                panel.classList.add('translate-x-full');
                openBtn.classList.remove('hidden');
            });
            openBtn.addEventListener('click', () => {
                panel.classList.remove('translate-x-full');
                openBtn.classList.add('hidden');
            });
        }

        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }

        function animate() {
            requestAnimationFrame(animate);
            controls.update();
            renderer.render(scene, camera);
        }
    </script>
