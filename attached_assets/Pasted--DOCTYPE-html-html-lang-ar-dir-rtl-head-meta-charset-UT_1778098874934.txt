    <script>
        let scene, camera, renderer, controls;
        let bagGroup, bagMesh, printMesh;
        let printTexture;
        let selectedColor = '#ffffff';
        let uploadedImageObj = null;
        let ambientLight, dirLight, floorMat;

        const SCALE = 10;

        init();
        animate();

        function init() {
            const container = document.getElementById('canvas-container');

            scene = new THREE.Scene();
            scene.background = new THREE.Color('#e2e8f0');

            camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 100);
            camera.position.set(5, 3, 7);

            renderer = new THREE.WebGLRenderer({ antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
            renderer.shadowMap.enabled = true;
            renderer.shadowMap.type = THREE.PCFSoftShadowMap;
            container.appendChild(renderer.domElement);

            controls = new THREE.OrbitControls(camera, renderer.domElement);
            controls.enableDamping = true;
            controls.dampingFactor = 0.05;
            controls.minDistance = 2;
            controls.maxDistance = 15;

            setupLighting();
            setupFloor();
            
            bagGroup = new THREE.Group();
            scene.add(bagGroup);

            updatePrintCanvas();
            updateBagGeometry();
            setupUIListeners();
            window.addEventListener('resize', onWindowResize);
        }

        function setupLighting() {
            ambientLight = new THREE.AmbientLight(0xffffff, 0.7);
            scene.add(ambientLight);

            dirLight = new THREE.DirectionalLight(0xffffff, 1.2);
            dirLight.position.set(4, 8, 6);
            dirLight.castShadow = true;
            dirLight.shadow.mapSize.width = 1024;
            dirLight.shadow.mapSize.height = 1024;
            scene.add(dirLight);

            const backLight = new THREE.PointLight(0xf8fafc, 0.5);
            backLight.position.set(-4, 4, -4);
            scene.add(backLight);
        }

        function setupFloor() {
            const floorGeo = new THREE.PlaneGeometry(30, 30);
            floorMat = new THREE.MeshStandardMaterial({ color: '#cbd5e1', roughness: 0.8 });
            const floor = new THREE.Mesh(floorGeo, floorMat);
            floor.rotation.x = -Math.PI / 2;
            floor.receiveShadow = true;
            scene.add(floor);
        }

        function getPlasticMaterial(type, colorHex, thickness) {
            let opts = {
                color: new THREE.Color(colorHex),
                side: THREE.DoubleSide,
                transparent: true,
            };

            // تأثير السماكة (35 إلى 150)
            const thicknessRatio = (thickness - 35) / (150 - 35); // 0 to 1

            if (type === 'ldpe') {
                opts.roughness = 0.15;
                opts.metalness = 0.1;
                // السماكة تزيد العتامة وتقلل الشفافية
                opts.transmission = 0.8 - (thicknessRatio * 0.4); 
                opts.opacity = 0.6 + (thicknessRatio * 0.35);
            } else { // HDPE
                opts.roughness = 0.6;
                opts.metalness = 0.0;
                opts.transmission = 0.0;
                opts.opacity = 0.9 + (thicknessRatio * 0.1);
            }
            return new THREE.MeshPhysicalMaterial(opts);
        }

        function updatePrintCanvas() {
            const canvas = document.getElementById('print-canvas');
            const ctx = canvas.getContext('2d');
            
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            if (uploadedImageObj) {
                // رسم الصورة المرفقة في المنتصف
                const imgAspect = uploadedImageObj.width / uploadedImageObj.height;
                let drawW = canvas.width * 0.6;
                let drawH = drawW / imgAspect;
                
                if (drawH > canvas.height * 0.6) {
                    drawH = canvas.height * 0.6;
                    drawW = drawH * imgAspect;
                }
                
                ctx.drawImage(uploadedImageObj, (canvas.width - drawW)/2, (canvas.height - drawH)/2, drawW, drawH);
            } else {
                // نص الفلكسو
                const text = document.getElementById('print-text').value;
                if(text) {
                    ctx.textAlign = 'center';
                    ctx.textBaseline = 'middle';
                    const cx = canvas.width / 2;
                    const cy = canvas.height / 2;

                    ctx.font = 'bold 100px Tajawal, sans-serif';
                    // لون الطباعة يعتمد على العكس لتوضيحه (كمثال)
                    ctx.fillStyle = '#0f172a';
                    ctx.fillText(text, cx, cy);
                    
                    ctx.strokeStyle = '#0ea5e9';
                    ctx.lineWidth = 4;
                    ctx.strokeText(text, cx+5, cy+5);
                }
            }

            if(!printTexture) {
                printTexture = new THREE.CanvasTexture(canvas);
                printTexture.anisotropy = renderer.capabilities.getMaxAnisotropy();
            } else {
                printTexture.needsUpdate = true;
            }
        }

        function updateBagGeometry() {
            while(bagGroup.children.length > 0){ 
                bagGroup.remove(bagGroup.children[0]); 
            }

            const w_cm = parseFloat(document.getElementById('width').value);
            const h_inches = parseFloat(document.getElementById('height').value);
            const d_cm = parseFloat(document.getElementById('depth').value);
            const thickness = parseFloat(document.getElementById('thickness').value);
            
            // تحويل البوصة إلى سم للطول
            const h_cm = h_inches * 2.54; 

            // تحويل للبرنامج
            const w = w_cm / SCALE;
            const h = h_cm / SCALE;
            let d = d_cm / SCALE;
            if (d === 0) d = 0.02; // منع الاختفاء

            const handleType = document.getElementById('handle').value;
            const plasticType = document.getElementById('type').value;

            const material = getPlasticMaterial(plasticType, selectedColor, thickness);

            // 1. هندسة الكيس
            let geo = new THREE.BoxGeometry(w, h, d, 32, 32, 4);
            geo.translate(0, h/2, 0);

            // حساب ارتفاع العلاقي بناءً على الطول الفعلي (10 إلى 15 سم)
            let vestHandleHeightCm = Math.min(15, Math.max(10, h_cm * 0.25));
            let vestHandleUnits = vestHandleHeightCm / SCALE;

            const pos = geo.getAttribute('position');
            for(let i = 0; i < pos.count; i++) {
                let x = pos.getX(i);
                let y = pos.getY(i);
                let z = pos.getZ(i);
                const yNorm = y / h;

                // إدخال الطية الجانبية
                if (Math.abs(x) > (w/2 - 0.05) && Math.abs(z) < (d/4)) {
                    x -= Math.sign(x) * (d * 0.4);
                }

                if (handleType === 'vest') {
                    // تشكيل مقبض العلاقي: تفريغ الجزء العلوي الأوسط
                    // إذا كنا في الجزء العلوي المخصص للمقبض (مثلاً أعلى 20%)
                    const handleThreshold = (h - vestHandleUnits) / h;
                    
                    if (yNorm > handleThreshold) {
                        // إذا كنا في الثلث الأوسط عرضياً، نخفض الارتفاع
                        if (Math.abs(x) < w * 0.25) {
                            // دالة جيبية لعمل تقوس في القص
                            const dip = vestHandleUnits * Math.cos((x / (w*0.25)) * (Math.PI/2));
                            y -= dip;
                        } else {
                            // الأطراف (المقابض) تنضم لبعضها قليلاً في الأعلى
                            x *= 0.9;
                        }
                    }
                } else {
                    // الأكياس العادية تُغلق من الأعلى قليلاً
                    if (yNorm > 0.85) {
                       z *= 0.2; 
                    }
                }

                pos.setXYZ(i, x, y, z);
            }
            geo.computeVertexNormals();

            bagMesh = new THREE.Mesh(geo, material);
            bagMesh.castShadow = true;
            bagGroup.add(bagMesh);

            // المقابض الإضافية
            if (handleType === 'banana') {
                const holeGeo = new THREE.TorusGeometry(w * 0.12, 0.02, 16, 32);
                holeGeo.scale(1, 0.35, 1);
                const holeMat = new THREE.MeshBasicMaterial({ color: 0x1e293b, transparent:true, opacity:0.5 }); 
                
                const hole = new THREE.Mesh(holeGeo, holeMat);
                hole.position.set(0, h * 0.85, 0); // في المنتصف
                bagGroup.add(hole);
            } 
            else if (handleType === 'loop') {
                const handleGeo = new THREE.TorusGeometry(w * 0.2, 0.04, 16, 32, Math.PI);
                const h1 = new THREE.Mesh(handleGeo, material);
                h1.position.set(0, h * 0.98, d/2);
                const h2 = new THREE.Mesh(handleGeo, material);
                h2.position.set(0, h * 0.98, -d/2);
                bagGroup.add(h1, h2);
            }

            // 3. طبقة الطباعة
            const printMat = new THREE.MeshBasicMaterial({
                map: printTexture,
                transparent: true,
                depthWrite: false,
                polygonOffset: true,
                polygonOffsetFactor: -1
            });
            
            const printGeo = geo.clone();
            printMesh = new THREE.Mesh(printGeo, printMat);
            printMesh.scale.set(1.002, 1.002, 1.002);
            bagGroup.add(printMesh);

            controls.target.set(0, h/2, 0);
        }

        function setupUIListeners() {
            const widthInput = document.getElementById('width');
            const depthInput = document.getElementById('depth');
            
            // قيد الطية (العمق) إلى 50% من العرض
            widthInput.addEventListener('input', (e) => {
                const w = parseFloat(e.target.value);
                document.getElementById('width-val').innerText = w;
                
                const maxD = Math.floor(w / 2);
                depthInput.max = maxD;
                if(parseFloat(depthInput.value) > maxD) {
                    depthInput.value = maxD;
                    document.getElementById('depth-val').innerText = maxD;
                }
                updateBagGeometry();
            });

            ['height', 'depth', 'thickness'].forEach(id => {
                document.getElementById(id).addEventListener('input', (e) => {
                    document.getElementById(`${id}-val`).innerText = e.target.value;
                    updateBagGeometry();
                });
            });

            ['handle', 'type'].forEach(id => {
                document.getElementById(id).addEventListener('change', updateBagGeometry);
            });

            // لوحة الألوان
            document.querySelectorAll('.color-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    document.querySelectorAll('.color-btn').forEach(b => b.classList.remove('active'));
                    e.target.classList.add('active');
                    selectedColor = e.target.getAttribute('data-color');
                    updateBagGeometry();
                });
            });

            // إرفاق الصورة
            const imgLoader = new Image();
            document.getElementById('design-upload').addEventListener('change', function(e) {
                const file = e.target.files[0];
                if (file) {
                    const reader = new FileReader();
                    reader.onload = function(evt) {
                        imgLoader.onload = function() {
                            uploadedImageObj = imgLoader;
                            updatePrintCanvas();
                        }
                        imgLoader.src = evt.target.result;
                    }
                    reader.readAsDataURL(file);
                }
            });

            document.getElementById('update-print-btn').addEventListener('click', updatePrintCanvas);

            // التحكم في الإضاءة والأرضية
            document.getElementById('light-intensity').addEventListener('input', (e) => {
                const val = parseFloat(e.target.value);
                document.getElementById('light-val').innerText = val;
                
                // مقياس الشدة بناءً على القيمة الافتراضية
                const scale = val / 100;
                if(ambientLight) ambientLight.intensity = 0.7 * scale;
                if(dirLight) dirLight.intensity = 1.2 * scale;
            });

            document.getElementById('floor-color').addEventListener('input', (e) => {
                const newColor = e.target.value;
                if(floorMat) {
                    floorMat.color.set(newColor);
                    scene.background.set(newColor); // تغيير لون الخلفية أيضاً لعمل مظهر استوديو متصل
                }
            });

            // إظهار/إخفاء القائمة
            const panel = document.getElementById('control-panel');
            const openBtn = document.getElementById('toggle-panel-btn');
            document.getElementById('close-panel-btn').addEventListener('click', () => {
                panel.classList.add('translate-x-full');
                openBtn.classList.remove('hidden');
            });
            openBtn.addEventListener('click', () => {
                panel.classList.remove('translate-x-full');
                openBtn.classList.add('hidden');
            });
        }

        function onWindowResize() {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        }

        function animate() {
            requestAnimationFrame(animate);
            controls.update();
            renderer.render(scene, camera);
        }
    </script>
