    <script type="module">
        import * as THREE from 'three';
        import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

        // --- Data Definitions ---
        const BAG_COLORS = {
            'أزرق': { hex: '#0047AB', roughness: 0.4, transmission: 0, metalness: 0.1 },
            'سماوي': { hex: '#87CEEB', roughness: 0.4, transmission: 0, metalness: 0.1 },
            'شفاف': { hex: '#FFFFFF', roughness: 0.1, transmission: 0.95, opacity: 1, transparent: true, ior: 1.5, thickness: 0.1 },
            'ثلجي': { hex: '#FAFAFA', roughness: 0.6, transmission: 0.8, opacity: 1, transparent: true, ior: 1.2, thickness: 0.5 },
            'أبيض': { hex: '#FFFFFF', roughness: 0.4, transmission: 0, metalness: 0 },
            'أصفر': { hex: '#FFD700', roughness: 0.4, transmission: 0, metalness: 0.1 },

            'أخضر': { hex: '#008000', roughness: 0.4, transmission: 0, metalness: 0.1 },
            'ذهبي': { hex: '#D4AF37', roughness: 0.25, transmission: 0, metalness: 0.8 },
            'رمادي': { hex: '#808080', roughness: 0.5, transmission: 0, metalness: 0.1 },
            'بني': { hex: '#8B4513', roughness: 0.6, transmission: 0, metalness: 0.1 },
            'فضي': { hex: '#C0C0C0', roughness: 0.2, transmission: 0, metalness: 0.9 },
            'بيج': { hex: '#F5F5DC', roughness: 0.5, transmission: 0, metalness: 0 },
            'عاجي': { hex: '#FFFFF0', roughness: 0.5, transmission: 0, metalness: 0 },
            'أحمر': { hex: '#DC143C', roughness: 0.4, transmission: 0, metalness: 0.1 },
            'وردي': { hex: '#FFC0CB', roughness: 0.4, transmission: 0, metalness: 0 },
            'تركوازي': { hex: '#40E0D0', roughness: 0.4, transmission: 0, metalness: 0.1 },
            'تفاحي': { hex: '#8DB600', roughness: 0.4, transmission: 0, metalness: 0.1 }
        };

        // --- State ---
        const state = {
            type: 'tshirt',
            w: 30,
            h: 40,
            g: 10,
            colorName: 'أزرق',
            colorsSide1: 2,
            colorsSide2: 1,
            printMode: 'text',
            printText: 'علامتك التجارية',
            printColor: '#000000',
            printSize: 80,
            printImgSize: 150
        };

        // --- Three.js Setup ---
        const container = document.getElementById('canvas-container');
        const scene = new THREE.Scene();
        // background is transparent to show css gradient
        
        const camera = new THREE.PerspectiveCamera(45, container.clientWidth / container.clientHeight, 1, 1000);
        camera.position.set(60, 40, 80);

        const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
        renderer.setSize(container.clientWidth, container.clientHeight);
        renderer.setPixelRatio(window.devicePixelRatio);
        renderer.shadowMap.enabled = true;
        renderer.shadowMap.type = THREE.PCFSoftShadowMap;
        container.appendChild(renderer.domElement);

        const controls = new OrbitControls(camera, renderer.domElement);
        controls.enableDamping = true;
        controls.dampingFactor = 0.05;
        controls.minDistance = 30;
        controls.maxDistance = 200;
        controls.target.set(0, 0, 0);

        // --- Lighting ---
        const ambientLight = new THREE.AmbientLight(0xffffff, 0.6);
        scene.add(ambientLight);

        const dirLight = new THREE.DirectionalLight(0xffffff, 1.2);
        dirLight.position.set(50, 60, 40);
        dirLight.castShadow = true;
        dirLight.shadow.mapSize.width = 2048;
        dirLight.shadow.mapSize.height = 2048;
        dirLight.shadow.bias = -0.001;
        scene.add(dirLight);

        const backLight = new THREE.DirectionalLight(0xffffff, 0.5);
        backLight.position.set(-50, 30, -50);
        scene.add(backLight);

        // Environment map for reflections (simulated with hemisphere)
        const hemiLight = new THREE.HemisphereLight(0xffffff, 0x444444, 0.6);
        hemiLight.position.set(0, 100, 0);
        scene.add(hemiLight);

        // --- Objects ---
        let bagGroup = null;
        let bagMaterial = new THREE.MeshPhysicalMaterial({
            side: THREE.DoubleSide,
            clearcoat: 0.5,
            clearcoatRoughness: 0.2
        });

        // --- Print Canvas Setup ---
        const printCanvas = document.createElement('canvas');
        printCanvas.width = 1024;
        printCanvas.height = 1024;
        const printCtx = printCanvas.getContext('2d');
        const printTexture = new THREE.CanvasTexture(printCanvas);
        printTexture.colorSpace = THREE.SRGBColorSpace;
        
        const printMaterial = new THREE.MeshPhysicalMaterial({
            map: printTexture,
            transparent: true,
            opacity: 0.85, // السماح للون الكيس الأساسي بالتأثير على لون الطباعة لتبدو واقعية (كالحبر)
            polygonOffset: true,
            polygonOffsetFactor: -1,
            depthWrite: false
        });

        // Measurement Lines
        const lineMaterial = new THREE.LineBasicMaterial({ color: 0x333333, linewidth: 2 });
        let wLine, hLine, gLine;
        
        function createMeasurementLines() {
            if(wLine) scene.remove(wLine, hLine, gLine);
            
            const wGeo = new THREE.BufferGeometry();
            wGeo.setAttribute('position', new THREE.BufferAttribute(new Float32Array(6), 3));
            wLine = new THREE.Line(wGeo, lineMaterial);
            
            const hGeo = new THREE.BufferGeometry();
            hGeo.setAttribute('position', new THREE.BufferAttribute(new Float32Array(6), 3));
            hLine = new THREE.Line(hGeo, lineMaterial);
            
            const gGeo = new THREE.BufferGeometry();
            gGeo.setAttribute('position', new THREE.BufferAttribute(new Float32Array(6), 3));
            gLine = new THREE.Line(gGeo, lineMaterial);
            
            scene.add(wLine, hLine, gLine);
        }

        // --- Geometry Builders ---
        function buildBag() {
            if (bagGroup) {
                scene.remove(bagGroup);
                bagGroup.traverse(child => {
                    if (child.isMesh) child.geometry.dispose();
                });
            }

            bagGroup = new THREE.Group();
            scene.add(bagGroup);

            const shape = new THREE.Shape();
            const hw = state.w / 2;
            const hh = state.h / 2;
            const d = state.g > 0 ? state.g : 0.2; // إعطاء عمق وهمي بسيط جداً إذا كانت الطية صفر ليظهر الكيس واقعياً
            const hd = d / 2;
            
            // Base Rectangle for all
            shape.moveTo(-hw, -hh);
            shape.lineTo(hw, -hh);

            const handleHeight = 10;
            const handleWidth = state.w * 0.25;

            if (state.type === 'tshirt') {
                // T-Shirt Cutout
                shape.lineTo(hw, hh + handleHeight); // Right outer
                shape.lineTo(hw - handleWidth, hh + handleHeight); // Right inner top
                
                // شكل العلاقة المستقيمة I__I
                shape.lineTo(hw - handleWidth, hh); // النزول لأسفل بشكل مستقيم
                shape.lineTo(-hw + handleWidth, hh); // خط أفقي في المنتصف (أرضية القص)
                shape.lineTo(-hw + handleWidth, hh + handleHeight); // الصعود لأعلى بشكل مستقيم

                shape.lineTo(-hw, hh + handleHeight); // Left outer top
                shape.lineTo(-hw, -hh); // Back to start

            } else {
                shape.lineTo(hw, hh);
                shape.lineTo(-hw, hh);
                shape.lineTo(-hw, -hh);
            }

            if (state.type === 'diecut') {
                // Oval Cutout
                const holePath = new THREE.Path();
                const holeW = Math.min(state.w * 0.3, 12) / 2;
                const holeH = 3;
                const holeY = hh - 6;
                holePath.absellipse(0, holeY, holeW, holeH, 0, Math.PI * 2, false, 0);
                shape.holes.push(holePath);
            }

            // --- بناء الكيس المفرغ من الداخل ---
            
            const shapeGeo = new THREE.ShapeGeometry(shape, 32);
            
            // 1. الوجه الأمامي
            const frontMesh = new THREE.Mesh(shapeGeo, bagMaterial);
            frontMesh.position.z = hd;
            frontMesh.castShadow = true; frontMesh.receiveShadow = true;
            bagGroup.add(frontMesh);

            // 1.5. طبقة الطباعة (تأخذ نفس شكل الوجه الأمامي تماماً)
            const printMesh = new THREE.Mesh(shapeGeo, printMaterial);
            printMesh.position.z = hd; // يتم حل التداخل بفضل خاصية polygonOffset
            bagGroup.add(printMesh);

            // 2. الوجه الخلفي
            const backMesh = new THREE.Mesh(shapeGeo, bagMaterial);
            backMesh.position.z = -hd;
            backMesh.rotation.y = Math.PI; // عكس الوجه للخلف
            backMesh.castShadow = true; backMesh.receiveShadow = true;
            bagGroup.add(backMesh);

            // دالة مساعدة لإنشاء الجدران (الطيات الجانبية والقاعدة)
            function addWall(w, h, px, py, pz, rx, ry) {
                const geo = new THREE.PlaneGeometry(w, h);
                const mesh = new THREE.Mesh(geo, bagMaterial);
                mesh.position.set(px, py, pz);
                if(rx) mesh.rotation.x = rx;
                if(ry) mesh.rotation.y = ry;
                mesh.castShadow = true; mesh.receiveShadow = true;
                bagGroup.add(mesh);
            }

            // 3. قاعدة الكيس (من الأسفل)
            addWall(state.w, d, 0, -hh, 0, Math.PI/2, 0);

            // 4. الجدران الجانبية والطيات
            if (state.type === 'tshirt') {
                const sideHeight = state.h + handleHeight;
                const centerY = (-hh + (hh + handleHeight)) / 2;
                
                // الطية اليسرى واليمنى
                addWall(d, sideHeight, -hw, centerY, 0, 0, Math.PI/2);
                addWall(d, sideHeight, hw, centerY, 0, 0, Math.PI/2);

                // إغلاق المقابض من الأعلى
                addWall(handleWidth, d, -hw + handleWidth/2, hh + handleHeight, 0, Math.PI/2, 0);
                addWall(handleWidth, d, hw - handleWidth/2, hh + handleHeight, 0, Math.PI/2, 0);

                // إغلاق الجدران الداخلية لمنطقة القص (I__I) لتبدو واقعية من الجانب
                addWall(d, handleHeight, hw - handleWidth, hh + handleHeight/2, 0, 0, Math.PI/2); // الجدار الداخلي الأيمن للمقبض
                addWall(d, handleHeight, -hw + handleWidth, hh + handleHeight/2, 0, 0, Math.PI/2); // الجدار الداخلي الأيسر للمقبض
                addWall(state.w - (handleWidth * 2), d, 0, hh, 0, Math.PI/2, 0); // الخط الأفقي المغلق بين المقبضين

            } else {
                addWall(d, state.h, -hw, 0, 0, 0, Math.PI/2);
                addWall(d, state.h, hw, 0, 0, 0, Math.PI/2);
            }

            // Add Soft Loops if needed
            if (state.type === 'softloop') {
                const loopRadius = state.w * 0.15;
                const tubeRadius = 0.6;
                const loopGeo = new THREE.TorusGeometry(loopRadius, tubeRadius, 8, 32, Math.PI);
                
                const handleFront = new THREE.Mesh(loopGeo, bagMaterial);
                handleFront.position.set(0, hh, hd);
                handleFront.castShadow = true;
                
                const handleBack = new THREE.Mesh(loopGeo, bagMaterial);
                handleBack.position.set(0, hh, -hd);
                handleBack.castShadow = true;

                bagGroup.add(handleFront);
                bagGroup.add(handleBack);
            }

            updateLines();
            applyColorMaterial();
            if(typeof updatePrintCanvas === 'function') updatePrintCanvas();
        }

        function updateLines() {
            if(!wLine) createMeasurementLines();
            
            const hw = state.w / 2;
            const hh = state.h / 2;
            const d = state.g > 0 ? state.g : 0.5;
            const hd = d / 2;
            const offset = 3; // distance from bag

            // Width Line (Bottom)
            const wPos = wLine.geometry.attributes.position.array;
            wPos[0] = -hw; wPos[1] = -hh - offset; wPos[2] = hd;
            wPos[3] = hw;  wPos[4] = -hh - offset; wPos[5] = hd;
            wLine.geometry.attributes.position.needsUpdate = true;

            // Height Line (Left Side)
            const hPos = hLine.geometry.attributes.position.array;
            hPos[0] = -hw - offset; hPos[1] = -hh; hPos[2] = hd;
            hPos[3] = -hw - offset; hPos[4] = hh;  hPos[5] = hd;
            hLine.geometry.attributes.position.needsUpdate = true;

            // Gusset Line (Bottom Right Depth)
            if(state.g > 0) {
                gLine.visible = true;
                const gPos = gLine.geometry.attributes.position.array;
                gPos[0] = hw + offset; gPos[1] = -hh; gPos[2] = hd;
                gPos[3] = hw + offset; gPos[4] = -hh; gPos[5] = -hd;
                gLine.geometry.attributes.position.needsUpdate = true;
            } else {
                gLine.visible = false;
            }
        }

        function applyColorMaterial() {
            const cProps = BAG_COLORS[state.colorName];
            if(!cProps) return;

            bagMaterial.color.set(cProps.hex);
            bagMaterial.roughness = cProps.roughness;
            bagMaterial.metalness = cProps.metalness !== undefined ? cProps.metalness : 0;
            
            // Handle Translucency / Clear
            if (cProps.transparent) {
                bagMaterial.transparent = true;
                bagMaterial.transmission = cProps.transmission;
                bagMaterial.opacity = cProps.opacity;
                bagMaterial.ior = cProps.ior;
                bagMaterial.thickness = cProps.thickness;
            } else {
                bagMaterial.transparent = false;
                bagMaterial.transmission = 0;
                bagMaterial.opacity = 1;
            }
            bagMaterial.needsUpdate = true;

            // تحديث خصائص مادة الطباعة لتتأثر بإضاءة وانعكاس لون الكيس
            printMaterial.roughness = cProps.roughness;
            printMaterial.metalness = cProps.metalness !== undefined ? cProps.metalness : 0;
            printMaterial.needsUpdate = true;
        }

        // --- 5. Print Logic (Text & Image) ---
        let loadedImage = null;

        function updatePrintCanvas() {
            printCtx.clearRect(0, 0, 1024, 1024);

            // إزاحة الطباعة لأسفل قليلاً في حال كان الكيس علاق لتجنب الطباعة فوق منطقة القص
            let yOffset = state.type === 'tshirt' ? 620 : 512;

            if (state.printMode === 'text') {
                if (state.printText.trim() !== '') {
                    printCtx.fillStyle = state.printColor;
                    printCtx.font = `bold ${state.printSize * 2}px Tajawal, sans-serif`;
                    printCtx.textAlign = 'center';
                    printCtx.textBaseline = 'middle';
                    printCtx.fillText(state.printText, 512, yOffset);
                }
            } else if (state.printMode === 'image' && loadedImage) {
                const maxDim = state.printImgSize * 4;
                let w = loadedImage.width;
                let h = loadedImage.height;
                const scale = Math.min(maxDim / w, maxDim / h);
                w *= scale;
                h *= scale;
                
                printCtx.drawImage(loadedImage, 512 - w/2, yOffset - h/2, w, h);

                // خوارزمية إزالة الخلفية البيضاء لمحاكاة الشفافية
                const imgData = printCtx.getImageData(0, 0, 1024, 1024);
                const data = imgData.data;
                for (let i = 0; i < data.length; i += 4) {
                    // إذا كان البكسل قريباً جداً من اللون الأبيض، اجعله شفافاً بالكامل (Alpha = 0)
                    if (data[i] > 235 && data[i+1] > 235 && data[i+2] > 235) {
                        data[i+3] = 0; 
                    }
                }
                printCtx.putImageData(imgData, 0, 0);
            }

            printTexture.needsUpdate = true;
        }
